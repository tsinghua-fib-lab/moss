413    def get_lane_waiting_at_end_vehicle_counts(
414        self, speed_threshold: float = 0.1, distance_to_end: float = 100
415    ) -> Dict[int, int]:
416        """
417        Get the number of vehicles of each lane with speed lower than `speed_threshold` and distance to end lower than `distance_to_end`
418
419        Returns:
420        - Dict: lane id -> number of vehicles
421        """
422
423        persons = self.fetch_persons()
424        lane_id = persons["lane_id"]
425        status = persons["status"]
426        v = persons["v"]
427        s = persons["s"]
428        filter = (status == PersonStatus.DRIVING.value) & (v < speed_threshold)
429        filtered_lane_id = lane_id[filter]
430        filtered_s = s[filter]
431        # find the distance to the end of the lane
432        lane_ids_for_count = []
433        for i, s in zip(filtered_lane_id, filtered_s):
434            if self.id2lanes[i].length - s < distance_to_end:
435                lane_ids_for_count.append(i)
436        # count for the lane id
437        unique, counts = np.unique(lane_ids_for_count, return_counts=True)
438        return dict(zip(unique, counts))
